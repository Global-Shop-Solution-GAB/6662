SELECT V_JOB_HEADER.JOB AS "Job", V_JOB_HEADER.SUFFIX AS "Suffix", IF(V_JOB_HEADER.FLAG_WO_PRTD = 'Y' AND V_JOB_HEADER.FLAG_WO_RELEASED = 'Y', 'R', IF(V_JOB_HEADER.FLAG_WO_RELEASED = 'Y', 'X', IF(V_JOB_HEADER.FLAG_WO_PRTD = 'Y', 'Y', ''))) AS "Printed", V_JOB_HEADER.DESCRIPTION AS "Description", V_JOB_HEADER.PART AS "Part Number", GCG_6662_Part_Maint.PO_Part as "PO PART", V_JOB_HEADER.DATE_DUE AS "Date Due" FROM V_JOB_HEADER LEFT JOIN GCG_6662_Part_Maint on V_JOB_HEADER.Part = GCG_6662_Part_Maint.WO_Part  WHERE JOB > '' ORDER BY "Job", "Suffix"
